113 				hideDelay:300,

137 	init:function(options){
138 		this.initOptions(options);	
139 		this.createComponent();

142 	/**
143 	 * 创建组件
144 	 * @private
145 	 */
146 	createComponent:function(){
147 		var self = this;
148 		var container = $(self.container).first()
149 		,parentNodes = container.find(self.parent);
150 		
151 		$(self.child).each(function(i,el){
152 			$(el).css("display","none")
153 			     .css("position","absolute");
154 		});
155 		
156 		parentNodes.each(function(i,el){	
157 			var child = $(el).children(self.child).first();
158 			$(el).bind(self.showEvent,function(e){
159 				if(self._curItem){
160 					if(self.hideAnimate){
161 					   self._curItem.stop(false,true);
162 					}else{
163 					   self._curItem.stop();
164 					}
165 				 self._curItem = null;
166 				}
167 				self._onShow.call(self,e,$(el),child,self);
168 			}).bind(self.hideEvent,function(e){				
169 				if(self.hideDelay>0){					
170 					if(self.hideAnimate){
171 						self._curItem = child.animate(self.hideAnimate,self.hideDelay,function(){
172 							self._onHide.call(self,e,$(el),child,self);
173 							self._curItem = null;
174 					    });	
175 					}else{
176 						self._curItem = child.delay(self.hideDelay).hide(0,function(){
177 							self._onHide.call(self,e,$(el),child,self);
178 							self._curItem = null;
179 					    });						
180 					}
181 				}else{
182 				  self._onHide.call(self,e,$(el),child,self);
183 				}
184 			});
185 		});
186 	},
187 	
188 	_onShow:function(e,parent,child,comp){
189 		child.show();
190 		if(this.selectedCls){parent.addClass(this.selectedCls);}
191 		if(this.isPositionMenuItem){this.setMenuItemPosition(parent,child,comp);}
192 		this.onShow(parent,child,comp);
193 	},
194 	
195 	_onHide:function(e,parent,child,comp){
196 		  child.hide();
197 		  if(this.selectedCls){parent.removeClass(this.selectedCls);}
198 		  this.onHide(parent,child,comp);
199 	},
200 	
201 	/**
202 	 * 设置当前显示菜单项位置，可通过初始化重构
203 	 * @public
204 	 * @param {jQueryObject} parent 菜单父节点jquery对象
205 	 * @param {jQueryObject} child 菜单子节点jquery对象
206 	 * @param {FE.operation.module.NavMenu} comp 当前NavMenu对象
207 	 */
208 	setMenuItemPosition:function(parent,child,comp){
209 		var p_offset = parent.offset();
210        
211 		var top=0,left=0;
212 		var m = this.marginAlign;
213 		switch(this.getAlignType(parent,child,comp)){
214 		 case "lb":		  
215 		   top = p_offset.top + p_offset.outerWidth();
216 		   left = p_offset.left;
217 			if(m && m.left){left += m.left;}
218 			if(m && m.top){top += m.top;}			 
219 		   break;
220 		 default:/* 默认对齐方式为"rt" */  
221 		   top = p_offset.top;
222 		   left = p_offset.left + parent.outerWidth(); 
223 		   
224            var w_top =0,w_h =0,a_top=0;
225            
226            var alignEl = $(this.alignElement || this.container);
227            
228            if(alignEl.length>0){
229         	   a_top = alignEl.first().offset().top;       	   
230            }else{
231         	   alignEl = null;
232            }
233 
234            w_top = $(window).scrollTop();
235 		   w_h = $(window).height();
236 
237 		   if(m && m.left){left += m.left;}
238 			if(m && m.top){top += m.top;}	
239 		   
240 		   if( child.outerHeight()+top > w_top+w_h){
241 			   top -= (child.outerHeight()+top-w_top-w_h);
242 			   if(top<w_top){top=w_top;}			   
243 			   if(alignEl && top<a_top){top=a_top;}
244 		   }
245 		   
246 		}
247 		
248 		child.offset(this.getItemPosition({left:left,top:top},parent,child));
249 	},
250 	
251 	/**
252 	 * 得到菜单项级别.(级别由1开始递增)
253 	 * @public
254 	 * @param {jQueryObject} menuItem 菜单项jquery对象
255 	 * @returns {Number} 菜单级数
256 	 */
257 	getItemLevel:function(menuItem){
258 		if($(this.container).length>0){
259 			var lvl = 0;
260 			for(var p = menuItem.parent();p.length>0;p=p.parent()){
261 				if(p.is(this.container)){return lvl;}
262 				if(p.is(this.parent)){lvl++;}
263 			}
264 		}
265 		return -1;
266 	},
