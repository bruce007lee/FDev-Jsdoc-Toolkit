291 		/**
292 		 * 创建组件
293 		 * @private
294 		 */
295 		createComponent:function(){
296 			var self = this;
297 			var container = $(self.container).first();
298 			
299 			var cfgs = this._getConfig(container,this._dateKey_);
300 			
301 			$.extend(this,cfgs);
302 			
303 			var items = container.find(self.itemSelector);
304 
305 			this.ct = container;
306 			this.items = items;
307 			
308 			items.each(function(i,el){
309 				var j_el = $(el);
310 				j_el.bind(self.hoverEvent,function(e){
311 					  self._doHover(i,j_el,self,items,container);
312 					}).bind(self.unhoverEvent,function(e){
313 					  self._doUnHover(i,j_el,self,items,container);
314 				});
315 		    });
316 			
317 			if(this.isInitStyle){
318 				this._doUnHover(-1,null,self,items,container);
319 			}
320 			
321 			if(this.isAutoFitWidth){
322 				this.fitWidth(this.width);
323 			}
324 			
325 			this.onStartup(this);
326 		},

329 		/**
330 		 * 设置组件自适应制定宽度和自适应方式
331 		 * @public
332 		 * @param {Number} width 指定宽度（不设置则为容器宽度）
333 		 * @param {String} type 对齐方式
334 		 */
335 		fitWidth:function(width,type){
336 			var w = width || (this.ct.width());
337 			var t = type || this.autoFitType;
338 			
339 			var items = this.items,t_w = 0,winc = 0;
340 			
341 			if(t==="e"){
342 				var e_w = Math.floor(w/items.length);
343 				items.each(function(i,el){
344 					if(i==(items.length-1)){
345 						$(el).width(w-(items.length-1)*e_w);
346 					}else{
347 						$(el).width(e_w);
348 					}
349 					
350 				});				
351 			}else{			
352 				items.each(function(i,el){
353 					t_w += $(el).width();			
354 				});
355 				
356 				items.each(function(i,el){
357 					var jel = $(el);
358 					var i_w = 0;					
359 					i_w = i==(items.length-1)?(w - winc):(Math.floor(w*(jel.width())/t_w));					
360 					winc += i_w;	
361 					jel.width(i_w);
362 				});
363 			}
364 		},

366 		/**
367 		 * Set style class when item hover on.
368 		 * @private
369 		 * @param {Number} i
370 		 * @param {FE.ui.service.module.NavCrumb} crumb
371 		 * @param {jQueryObject} items
372 		 * @param {jQueryObject} ct
373 		 */
374 		_doHover:function(i,el,crumb,items,ct){			
375 			var selIdx = crumb.getSelectedIndex();
376 			
377 			items.each(function(s_i,s_el){
378 				var js_el = $(s_el);	
379 				crumb._clearClass(js_el);
380 				
381 				if(s_i<i && selIdx == s_i){
382 					js_el.addClass(crumb.prefixSelectedCls);					
383 					if(s_i==i-1 ){
384 						js_el.addClass(crumb.hoverPrevCls);
385 					}
386 				}else if(s_i>i && selIdx == s_i){
387 					js_el.addClass(crumb.postfixSelectedCls);
388 					if(s_i==i+1 ){
389 						js_el.addClass(crumb.hoverNextCls);
390 					}
391 				}else if(selIdx == s_i ){
392 					js_el.addClass(crumb.hoverSelectedCls);
393 				}else if(s_i==i+1 ){
394 					js_el.addClass(crumb.hoverNextCls);
395 				}else if(s_i==i-1){
396 					js_el.addClass(crumb.hoverPrevCls);
397 				}else if(s_i<i){
398 					js_el.addClass(crumb.hoverStepCls);
399 				}
400 				
401 				
402 				
403 				if(!el.hasClass(crumb.selectedCls)){
404 					el.addClass(crumb.hoverCls);
405 				}
406 			});
407 					
408 			crumb.onHover(crumb);
409 		},

428 					js_el.addClass(crumb.hoverPrevCls);
429 				}else if(s_i==selIdx+1){
430 					js_el.addClass(crumb.hoverNextCls);
431 				}else if(s_i==selIdx){
432 					js_el.addClass(crumb.defaultSelectedCls);
433 				}else if(s_i<selIdx){
434 					js_el.addClass(crumb.hoverStepCls);
435 				}
436 				
437 			});	
438 			
439 			crumb.onUnHover(crumb);
